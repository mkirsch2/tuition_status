SELECT previous_college.id,
	previous_college.program,
	previous_college.previous_college,
	age_gender_ethnicity.ethnic_description,
	age_gender_ethnicity.gender
INTO combined_table_1
FROM previous_college
INNER JOIN age_gender_ethnicity ON age_gender_ethnicity.student = previous_college.student;

SELECT combined_table_1.id,
	combined_table_1.program,
	combined_table_1.previous_college,
	combined_table_1.ethnic_description,
	combined_table_1.gender,
	student_summary_other_education.grad_date,
	student_summary_other_education.hs_ged,
	student_summary_other_education.hs_ged_grad_date
INTO combined_table_2
FROM combined_table_1
INNER JOIN student_summary_other_education ON student_summary_other_education.id = combined_table_1.id;

SELECT combined_table_2.id,
	combined_table_2.program,
	combined_table_2.previous_college,
	combined_table_2.ethnic_description,
	combined_table_2.gender,
	combined_table_2.grad_date,
	combined_table_2.hs_ged,
	combined_table_2.hs_ged_grad_date,
	age.date_of_birth
INTO combined_table_3
FROM combined_table_2
INNER JOIN age ON age.id = combined_table_2.id;

SELECT combined_table_3.id,
	combined_table_3.program,
	combined_table_3.previous_college,
	combined_table_3.ethnic_description,
	combined_table_3.gender,
	combined_table_3.grad_date,
	combined_table_3.hs_ged,
	combined_table_3.hs_ged_grad_date,
	combined_table_3.date_of_birth,
	final_attendance.hours_attended,
	final_attendance.hours_scheduled,
	final_attendance.attendance_percentage
INTO combined_table_4
FROM combined_table_3
INNER JOIN final_attendance ON final_attendance.id = combined_table_3.id;

SELECT combined_table_4.id,
	combined_table_4.program,
	combined_table_4.previous_college,
	combined_table_4.ethnic_description,
	combined_table_4.gender,
	combined_table_4.grad_date,
	combined_table_4.hs_ged,
	combined_table_4.hs_ged_grad_date,
	combined_table_4.date_of_birth,
	combined_table_4.hours_attended,
	combined_table_4.hours_scheduled,
	combined_table_4.attendance_percentage,
	final_gpas.gpa
INTO combined_table_5
FROM combined_table_4
INNER JOIN final_gpas ON final_gpas.id = combined_table_4.id;

SELECT combined_table_5.id,
	combined_table_5.program,
	combined_table_5.previous_college,
	combined_table_5.ethnic_description,
	combined_table_5.gender,
	combined_table_5.grad_date,
	combined_table_5.hs_ged,
	combined_table_5.hs_ged_grad_date,
	combined_table_5.date_of_birth,
	combined_table_5.hours_attended,
	combined_table_5.hours_scheduled,
	combined_table_5.attendance_percentage,
	combined_table_5.gpa,
	default_status.default_status
INTO combined_table_6
FROM combined_table_5
INNER JOIN default_status ON default_status.id = combined_table_5.id;

SELECT * FROM combined_table_6;
